var http = require('http');
var server = http.createServer(function(request, response){
    response.writeHead(200, {'content-type': 'text/plain' })
    response.write("empty\n");
    setTimeout(function(){
        response.end("site\n");
    }, 2000)
});
server.listen(8000);
